                categories {
                    int id PK
                    string name
                }
                products {
                    int id PK
                    string name
                    text description
                    double price
                    int category_id FK
                }
                customers {
                    int id PK
                    string name
                    string address
                    string phone
                }
                orders {
                    int id PK
                    datetime date
                    int customer_id FK
                    enum status
                    double total_amount
                }
                order_items {
                    int id PK
                    int order_id FK
                    int product_id FK
                    int quantity
                    double price
                }
                product_images {
                    int id PK
                    int product_id FK
                    text image_name
                    timestamp created_at
                }
